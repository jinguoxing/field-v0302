下面给你一个**可直接落你们前端技术栈**的 `ConfigEditor.tsx` 骨架（React 18 + TSX + Tailwind + React Router v7），把之前的 **Frame Spec / props / schema** 串起来：能跑通

* 左侧 Tab 切换
* 左侧规则列表（示例：Naming Rules）
* 中间表单编辑（示例：Weights、Naming Rule）
* 右侧即时仿真面板（mock simulate API）
* TopBar 的保存/校验/导入导出（mock）
* 修改即触发 debounce simulate（WHAT-IF）

你们只需要把 mock API 替换成真实接口，或把组件拆成你们自己的 UI 库即可。

> 说明：这是一个“单文件可运行骨架”，便于你们让大模型直接生成完整页面。真实项目建议按组件拆分。

function confidenceLevel(conf: number): ConfidenceLevel {

// debounce hook
function useDebouncedEffect(fn: () => void, deps: any[], delay: number) {
  const timeoutRef = useRef<number | null>(null);
  useEffect(() => {
    if (timeoutRef.current) window.clearTimeout(timeoutRef.current);
    timeoutRef.current = window.setTimeout(() => fn(), delay);
    return () => {
      if (timeoutRef.current) window.clearTimeout(timeoutRef.current);
    };
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, deps);
}

// ------------------------- UI Components (minimal) -------------------------
function Pill({ children, tone }: { children: React.ReactNode; tone?: "gray" | "green" | "yellow" | "red" }) {
  const bg =
    tone === "green" ? "bg-emerald-900/40 text-emerald-200 border-emerald-700/50"
    : tone === "yellow" ? "bg-amber-900/40 text-amber-200 border-amber-700/50"
    : tone === "red" ? "bg-rose-900/40 text-rose-200 border-rose-700/50"
    : "bg-slate-800 text-slate-200 border-slate-700";
  return <span className={cls("inline-flex items-center px-2 py-0.5 rounded-full border text-xs", bg)}>{children}</span>;
}

function SectionTitle({ title, desc }: { title: string; desc?: string }) {
  return (

      {desc && <div className="text-xs text-slate-400">{desc}</div>}

  );
}

function NumberField({ label, value, onChange, min, max, step }: { label: string; value: number; onChange: (v: number) => void; min?: number; max?: number; step?: number }) {
  return (
    <label className="flex items-center justify-between gap-3 py-1">

        type="number"

        min={min}
        max={max}
        step={step ?? 0.01}
        onChange={(e) => onChange(Number(e.target.value))}
      />
    </label>
  );
}

function Toggle({ label, value, onChange }: { label: string; value: boolean; onChange: (v: boolean) => void }) {
  return (
    <label className="flex items-center justify-between gap-3 py-1">
      <span className="text-xs text-slate-300">{label}</span>
      <button
        className={cls(
          "w-12 h-6 rounded-full border transition",
          value ? "bg-emerald-700/40 border-emerald-600" : "bg-slate-800 border-slate-700"
        )}
        onClick={() => onChange(!value)}
        type="button"
      >
        <span className={cls("block w-5 h-5 rounded-full bg-slate-200 transition translate-x-0.5", value && "translate-x-6")} />
      </button>
    </label>
  );
}

  onSelect,
  onToggleEnabled,

  onToggleEnabled: (id: string, enabled: boolean) => void;

        >
          <div className="flex items-center justify-between gap-2">
            <div className="text-xs font-semibold text-slate-100">{r.id}</div>

                className="text-xs text-slate-300 hover:text-slate-100"
                onClick={(e) => {
                  e.stopPropagation();
                  onToggleEnabled(r.id, !r.enabled);

                type="button"
              >
                {r.enabled ? "禁用" : "启用"}
              </button>
            </div>
          </div>
          <div className="mt-1 text-[11px] text-slate-400 truncate">{r.regex}</div>
        </button>
      ))}
    </div>
  );
}

              onChange={(v) => updateWeights("llm.max_delta_per_candidate", v)}
              min={0}
              max={0.05}
              step={0.01}
            />
          </div>

          <div className="rounded-2xl border border-slate-800 bg-slate-900/40 p-4">
            <SectionTitle title="Raw Editor (YAML/JSON)" desc="可选：展示/编辑原始配置（这里仅示例展示）。" />
            <pre className="text-[11px] text-slate-300 whitespace-pre-wrap">{JSON.stringify(config.weights, null, 2)}</pre>
          </div>
        </div>
      );
    }

    if (activeTab === "naming_rules" && selectedRule) {
      return (
        <div className="space-y-6">
          <div className="rounded-2xl border border-slate-800 bg-slate-900/40 p-4">
            <SectionTitle title="Rule Meta" desc="rule_id / enabled / priority / scope" />
            <Toggle label="enabled" value={selectedRule.enabled} onChange={(v) => updateNamingRule(selectedRule.id, { enabled: v })} />
            <NumberField label="priority" value={selectedRule.priority} onChange={(v) => updateNamingRule(selectedRule.id, { priority: v })} min={0} step={1} />
            <div className="text-xs text-slate-300 mt-2">scope: <Pill>{selectedRule.scope}</Pill></div>
          </div>

          <div className="rounded-2xl border border-slate-800 bg-slate-900/40 p-4">
            <SectionTitle title="Pattern" desc="regex 命中 → 产生 NAMING 证据与 delta" />
            <label className="block text-xs text-slate-300 mb-1">regex</label>
            <input
              className="w-full rounded-md bg-slate-900 border border-slate-700 px-2 py-2 text-xs text-slate-100"
              value={selectedRule.regex}
              onChange={(e) => updateNamingRule(selectedRule.id, { regex: e.target.value })}
            />
            <div className="mt-2 text-[11px] text-slate-400">提示：此处应做 regexCompile 校验与测试输入。</div>
          </div>

          <div className="rounded-2xl border border-slate-800 bg-slate-900/40 p-4">
            <SectionTitle title="Delta Mapping" desc="建议用 DeltaTable 组件编辑（此处简化展示 JSON）。" />
            <div className="grid grid-cols-2 gap-3">
              <div>
                <div className="text-xs text-slate-300 mb-1">type_delta</div>
                <pre className="text-[11px] text-slate-300 whitespace-pre-wrap">{JSON.stringify(selectedRule.type_delta, null, 2)}</pre>
              </div>
              <div>
                <div className="text-xs text-slate-300 mb-1">role_delta</div>
                <pre className="text-[11px] text-slate-300 whitespace-pre-wrap">{JSON.stringify(selectedRule.role_delta, null, 2)}</pre>
              </div>
            </div>
          </div>

          <div className="rounded-2xl border border-slate-800 bg-slate-900/40 p-4">
            <SectionTitle title="Evidence Template" desc="title/summary 模板用于 evidence 结构化输出。" />
            <label className="block text-xs text-slate-300 mb-1">title_template</label>
            <input
              className="w-full rounded-md bg-slate-900 border border-slate-700 px-2 py-2 text-xs text-slate-100"
              value={selectedRule.title_template ?? ""}
              onChange={(e) => updateNamingRule(selectedRule.id, { title_template: e.target.value })}
            />
            <label className="block text-xs text-slate-300 mb-1 mt-3">summary_template</label>
            <textarea
              className="w-full rounded-md bg-slate-900 border border-slate-700 px-2 py-2 text-xs text-slate-100"
              rows={2}
              value={selectedRule.summary_template ?? ""}
              onChange={(e) => updateNamingRule(selectedRule.id, { summary_template: e.target.value })}
            />
          </div>
        </div>
      );
    }

    return (
      <div className="rounded-2xl border border-slate-800 bg-slate-900/40 p-4 text-xs text-slate-400">
        该 Tab 的中间编辑区示例未展开。你可以按之前我给的 Frame Spec，把对应字段与组件补齐。
      </div>
    );
  }

  function renderRightPane() {
    const res = simMode === "BASELINE" ? baseline : whatIf;
    return (
      <div className="space-y-4">
        <div className="rounded-2xl border border-slate-800 bg-slate-900/40 p-4">
          <SectionTitle title="即时仿真" desc="编辑配置后自动 what-if 重算（防抖 300ms）。" />
          <div className="flex items-center gap-2 mb-2">
            <button
              className={cls("px-2 py-1 rounded-md text-xs border", simMode === "WHAT_IF" ? "border-emerald-600 text-emerald-200" : "border-slate-700 text-slate-300")}
              onClick={() => setSimMode("WHAT_IF")}
              type="button"
            >
              WHAT-IF
            </button>
            <button
              className={cls("px-2 py-1 rounded-md text-xs border", simMode === "BASELINE" ? "border-emerald-600 text-emerald-200" : "border-slate-700 text-slate-300")}
              onClick={() => setSimMode("BASELINE")}
              type="button"
            >
              BASELINE
            </button>
            {simLoading && <Pill tone="yellow">Simulating…</Pill>}
          </div>

          <div className="grid grid-cols-2 gap-2">
            <input
              className="rounded-md bg-slate-900 border border-slate-700 px-2 py-2 text-xs text-slate-100"
              value={sample.mode === "FIELD_ID" ? sample.fieldId : ""}
              placeholder="fieldId"
              onChange={(e) => setSample({ mode: "FIELD_ID", fieldId: e.target.value })}
            />
            <button
              className="rounded-md border border-slate-700 bg-slate-900/40 px-2 py-2 text-xs text-slate-200 hover:border-slate-600"
              onClick={async () => {
                setSimLoading(true);
                try {
                  const r = await apiSimulate(versionId, config as any, sample);
                  setWhatIf(r);
                } finally {
                  setSimLoading(false);
                }
              }}
              type="button"
            >
              Run
            </button>
          </div>

          {res && (
            <div className="mt-3 space-y-2">
              <div className="flex flex-wrap gap-2">
                <Pill tone={res.queue === "CONFLICT" ? "red" : res.queue === "AUTO_PASS" ? "green" : "yellow"}>{res.queue}</Pill>
                <Pill>{`completeness ${toPct(res.completeness)}`}</Pill>
                <Pill>{`gap ${toPct(res.gap12)}`}</Pill>
                <Pill>{`ignore ${toPct(res.ignoreScore)}`}</Pill>
              </div>

              <div className="rounded-xl border border-slate-800 bg-slate-950/30 p-3">
                <div className="text-xs font-semibold text-slate-100 mb-2">Joint Top3</div>
                <div className="space-y-1">
                  {res.jointTop3.map((c, i) => (
                    <div key={i} className="flex items-center justify-between text-[11px] text-slate-300">
                      <div>{i + 1}. {c.semanticType} + {c.role}</div>
                      <Pill tone={c.confidenceLevel === "HIGH" ? "green" : c.confidenceLevel === "MED" ? "yellow" : "gray"}>{toPct(c.confidence)}</Pill>
                    </div>
                  ))}
                </div>
              </div>

              <div className="space-y-2">
                <div className="text-xs font-semibold text-slate-100">Evidence Top3</div>
                {res.evidenceTop3.map((e) => <EvidenceRow key={e.evidenceId} e={e} />)}
              </div>
            </div>
          )}
        </div>

        <div className="rounded-2xl border border-slate-800 bg-slate-900/40 p-4">
          <SectionTitle title="Breakdown" desc="按维度/按规则贡献（可折叠做成表格）。" />
          {res?.breakdownByDimension?.map((b) => (
            <div key={b.key} className="flex items-center justify-between text-[11px] text-slate-300 py-1">
              <span>{b.key}</span>
              <Pill tone="green">{`+${toPct(b.contributionTop1)}`}</Pill>
            </div>
          ))}
        </div>
      </div>
    );
  }

  // ------------------------- Page -------------------------
  return (
    <div className="min-h-screen bg-slate-950 text-slate-100">
      {/* TopBar */}
      <div className="sticky top-0 z-20 border-b border-slate-800 bg-slate-950/80 backdrop-blur">
        <div className="mx-auto max-w-[1400px] px-4 py-3 flex items-center justify-between gap-3">
          <div className="flex items-center gap-3">
            <div className="text-sm font-semibold">配置编辑器</div>
            <Pill>{config.versionId}</Pill>
            <Pill tone={config.status === "DRAFT" ? "yellow" : "green"}>{config.status}</Pill>
            {config.baseVersionId && <Pill>{`base ${config.baseVersionId}`}</Pill>}
            {dirty && <Pill tone="red">Unsaved</Pill>}
          </div>
          <div className="flex items-center gap-2">
            <button className="rounded-md bg-emerald-600 px-3 py-1.5 text-xs font-semibold" onClick={onSave} type="button">
              保存
            </button>
            <button className="rounded-md border border-slate-700 px-3 py-1.5 text-xs" onClick={onValidate} type="button">
              校验
            </button>
            <button className="rounded-md border border-slate-700 px-3 py-1.5 text-xs" onClick={onDiff} type="button">
              生成Diff
            </button>
            <button className="rounded-md border border-rose-700 bg-rose-900/30 px-3 py-1.5 text-xs" onClick={onPublish} type="button">
              发布/灰度
            </button>
          </div>
        </div>
      </div>

      {/* Main */}
      <div className="mx-auto max-w-[1400px] px-4 py-4 grid grid-cols-12 gap-4">
        {/* Left Nav */}
        <div className="col-span-2">
          <div className="sticky top-[72px] space-y-2">
            {TABS.map((t) => (
              <button
                key={t.id}
                className={cls(
                  "w-full text-left rounded-xl border px-3 py-2 text-xs transition",
                  activeTab === t.id ? "border-emerald-600 bg-emerald-900/20" : "border-slate-800 bg-slate-900/40 hover:border-slate-700"
                )}
                onClick={() => setActiveTab(t.id)}
                type="button"
              >
                {t.title}
                <div className="mt-1 text-[11px] text-slate-400">{t.domain}</div>
              </button>
            ))}
          </div>
        </div>

        {/* Left Pane */}
        <div className="col-span-3">
          <div className="rounded-2xl border border-slate-800 bg-slate-900/30 p-3">
            {renderLeftPane()}
          </div>
        </div>

        {/* Center Pane */}
        <div className="col-span-4">
          {renderCenterPane()}
        </div>

        {/* Right Pane */}
        <div className="col-span-3">
          {renderRightPane()}
        </div>
      </div>
    </div>
  );
}
